===
Simple string variable
===

variable "image_id" {
  type = string
}

variable "create" {
  type = bool
}

variable "how_many" {
  type = number
}

---

(configuration
  (variable (variable_name) (variable_block
    (type (string_ty))))

  (variable (variable_name) (variable_block
    (type (bool_ty))))

  (variable (variable_name) (variable_block
    (type (number_ty)))))

Variable with default
===

  type    = string

}

  type    = list(number)
  default = [1, 2, 3]
}

(configuration
  (variable (variable_name) (variable_block
    (type (string_ty))
    (default (string_literal))))

    (default (list (number) (number) (number))))))
